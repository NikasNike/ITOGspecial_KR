CREATE TABLE smoll_animal (
	id INT PRIMARY KEY AUTO_INCREMENT,
	animal_name CHAR(30),
    commands TEXT,
    date DATE,
    age TEXT
);


DELIMITER $$
CREATE FUNCTION age_animal (date_s DATE)
RETURNS TEXT
DETERMINISTIC
BEGIN
    DECLARE res TEXT DEFAULT '';
	SET res = CONCAT(
            TIMESTAMPDIFF(YEAR, date_s, CURDATE()),
            ' years ',
            TIMESTAMPDIFF(MONTH, date_s, CURDATE()) % 12,
            ' month'
        );
	RETURN res;
END $$
